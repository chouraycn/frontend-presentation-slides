Core Implementation

The critical 12 lines

// Key insight: batch DB calls instead of N+1 queries
async function loadItems(ids: string[]) {
  const rows = await db.query(
    `SELECT * FROM items WHERE id = ANY($1)`,
    [ids]
  );
  return groupById(rows);
}
